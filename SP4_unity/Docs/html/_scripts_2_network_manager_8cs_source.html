567  private void SceneReady(Scene scene, LoadSceneMode mode)
568  {
569  // If we are loading a completely new scene then we will need
570  // to clear out all the old objects that were stored as they
571  // are no longer needed
572  if (mode != LoadSceneMode.Additive)
573  {
574  pendingObjects.Clear();
575  pendingNetworkObjects.Clear();
576  loadedScenes.Clear();
577  }
578  lock(NetworkObject.PendingCreatesLock)
579  {
580  loadingScenes.Remove(scene.buildIndex);
581  }
582  loadedScenes.Add(scene.buildIndex);
583 
584 
585  if (networkSceneLoaded != null)
586  networkSceneLoaded(scene, mode);
587 
588  BMSByte data = ObjectMapper.BMSByte(scene.buildIndex, (int)mode);
589 
590  Binary frame = new Binary(Networker.Time.Timestep, false, data, Networker is IServer ? Receivers.All : Receivers.Server, MessageGroupIds.VIEW_CHANGE, Networker is BaseTCP);
591 
592  // Send the binary frame to either the server or the clients
593  SendFrame(Networker, frame);
594 
595  // Go through all of the current NetworkBehaviors in the order that Unity finds them in
596  // and associate them with the id that the network will be giving them as a lookup
597  int currentAttachCode = 1;
598  var behaviors = FindObjectsOfType<NetworkBehavior>().Where(b => !b.Initialized)
599  .OrderBy(b => b.GetType().ToString())
600  .OrderBy(b => b.name)
601  .OrderBy(b => Vector3.Distance(Vector3.zero, b.transform.position))
602  .ToList();
603 
604  if (behaviors.Count == 0)
605  {
606  if (Networker is IClient)
607  {
608  if (loadingScenes.Count > 0)
609  NetworkObject.Flush(Networker, loadingScenes, CreatePendingObjects);
610  else
611  {
612  NetworkObject.Flush(Networker, loadingScenes);
613  if(pendingObjects.Count == 0)
614  Networker.objectCreated -= CreatePendingObjects;
615  }
616  }
617 
618 
619  return;
620  }
621 
622  foreach (NetworkBehavior behavior in behaviors)
623  {
624  behavior.TempAttachCode = scene.buildIndex << 16;
625  behavior.TempAttachCode += currentAttachCode++;
626  behavior.TempAttachCode = -behavior.TempAttachCode;
627  }
628 
629  if (Networker is IClient)
630  {
631  // This would occur if objects in the additive scene arrives at the same time as the
632  // "single" scene and were flushed.
633  if (mode == LoadSceneMode.Additive && pendingNetworkObjects.Count > 0)
634  {
635  NetworkObject foundNetworkObject;
636  for (int i = 0; i < behaviors.Count; i++)
637  {
638  if (pendingNetworkObjects.TryGetValue(behaviors[i].TempAttachCode, out foundNetworkObject))
639  {
640  behaviors[i].Initialize(foundNetworkObject);
641  pendingNetworkObjects.Remove(behaviors[i].TempAttachCode);
642  behaviors.RemoveAt(i--);
643  }
644  }
645  }
646 
647  foreach (NetworkBehavior behavior in behaviors)
648  pendingObjects.Add(behavior.TempAttachCode, behavior);
649 
650  NetworkObject.Flush(Networker, loadingScenes, CreatePendingObjects);
651 
652  if (pendingObjects.Count == 0 && loadingScenes.Count == 0)
653  Networker.objectCreated -= CreatePendingObjects;
654 
655  } else
656  {
657  // Go through all of the pending NetworkBehavior objects and initialize them on the network
658  foreach (INetworkBehavior behavior in behaviors)
659  behavior.Initialize(Networker);
660 
661  }
662  }
