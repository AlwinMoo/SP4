117  // Allocate new action queue by update type if there's no one exists.

136  private void HandleActions(UpdateType updateType)
137  {
138  // Allocate new action queue by update type if there's no one exists.
139  if (!actionQueueDict.ContainsKey(updateType))
140  {
141  actionQueueDict.Add(updateType, new Queue<Action>());
142 
143  // Since an action runner depends on the action queue, allocate new one here.
144  actionRunnerDict.Add(updateType, new Queue<Action>());
145 
146  }
147  Queue<Action> mainThreadActions = actionQueueDict[updateType];
148  Queue<Action> mainThreadActionsRunner = actionRunnerDict[updateType];
149 
150  lock (mainThreadActions)
151  {
152  // Flush the list to unlock the thread as fast as possible
153  if (mainThreadActions.Count > 0)
154  {
155  while (mainThreadActions.Count > 0)
156  mainThreadActionsRunner.Enqueue(mainThreadActions.Dequeue());
157  }
158  }
159 
160  // If there are any functions in the list, then run
161  // them all and then clear the list
162  if (mainThreadActionsRunner.Count > 0)
163  {
164  while (mainThreadActionsRunner.Count > 0)
165  mainThreadActionsRunner.Dequeue()();
166  }
167  }
