210  public void Update()

224  if (readFlushTimer > READ_FLUSH_TIME)
225  {
226  if (readUpdateId != previousReadUpdateId && readSamples != null && readSamples.Count > 0)
227  {
228  previousReadUpdateId = readUpdateId;
229 
230  lock (readSamples)
231  {
232  audio.clip = AudioClip.Create("VOIP", readSamples.Count, channels, Frequency, false);
233  audio.spatialBlend = audio3D;
234 
235  audio.clip.SetData(readSamples.ToArray(), 0);
236  if (!audio.isPlaying) audio.Play();
237 
238  readSamples.Clear();
239  }
240  }
241 
242  readFlushTimer = 0.0f;
243  }

248  private void VOIPWorker()

251  {

266  MainThreadManager.ThreadSleep(10);

289  private byte[] ToByteArray(List<float> sampleList)
290  {
291  int len = sampleList.Count * 4;
292  byte[] byteArray = new byte[len];
293  int pos = 0;
294 
295  for (int i = 0; i < sampleList.Count; i++)
296  {
297  byte[] data = BitConverter.GetBytes(sampleList[i]);
298  Array.Copy(data, 0, byteArray, pos, 4);
299  pos += 4;
300  }
301 
302  return byteArray;
303  }
