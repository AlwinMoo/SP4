243  private byte[] SerializeMap()
244  {
245  // Get the 3-dimensional lengths of the world
246  int width = (int)Mathf.Ceil((max.x - min.x)) + 1;
247  int height = (int)Mathf.Ceil((max.y - min.y)) + 1;
248  int depth = (int)Mathf.Ceil((max.z - min.z)) + 1;
249 
250  // Create a bit array to store the toggles for cubes
251  BitArray bitmap = new BitArray(width * height * depth);
252 
253  // pack the cubes into the bit array
254  int idx = 0;
255  foreach (GameObject obj in primitiveInstances.Values)
256  {
257  Vector3 d = obj.transform.position - min;
258  idx = (int)((d.z * width * height) + (d.y * width) + d.x);
259  bitmap[idx] = true;
260  }
261 
262  // Create a byte array to hold the packed bit data
263  byte[] mapData = new byte[(int)Mathf.Ceil(bitmap.Length / 8.0f)];
264 
265  // Copy the bit mapped data into the above byte array
266  ((ICollection)bitmap).CopyTo(mapData, 0);
267 
268  return mapData;
269  }

278  private void DeserializeMap(Vector3 minimum, Vector3 maximum, byte[] data)
279  {
280  min = minimum;

283  // Get the supplied 3-dimensional lengths of the world
284  int width = (int)Mathf.Ceil((max.x - min.x)) + 1;
285  int height = (int)Mathf.Ceil((max.y - min.y)) + 1;
286  //int depth = (int)Mathf.Ceil((max.z - min.z)) + 1;
287 
288  // Unpack the byte array into a bit array and begin mapping the world
289  int idx, z, y, x;

292  {
